SQL Clause

CREATE TABLE IF NOT EXISTS users() --- გამოიყენება მონაცემთა ცხრილის შესაქმნელათ

SELECT * FROM users --- გამოიყენება მონაცემების წამოსაღებათ

SELECT name, age FROM users --- გამოიყენება კონკრეტული სვეტის წამოსაღებათ

SELECT * FROM users WHERE age > 18 --- გამოიყენება რიგების გასაფილტრად და მათ წამოსაღებათ

AND, OR, NOT --- ლოგიკური ოპერატორები

INSERT INTO users (name, age, country) VALUES ('Luka', 16, 'Georgia') --- გამოიყენება ახალი მონაცემების დასამატებლად

DELETE FROM users WHERE age < 13 --- გამოიყენება მონაცემების წასაშლელად

SELECT * FROM users ORDER BY age ASC --- ალაგებს მონაცემების ზრდადობით (DESC --- კლებადობით)

SELECT * FROM users LIMIT 5 --- აბრუნებს მხოლოდ იმდენ მონაცემთა რიგს რამდენიც ლიმიტის შემდეგ წერია(5)

SELECT DISTINCT country FROM users --- აშორებს დუპლიკატებს

SELECT * FROM users WHERE name LIKE 'Lu%' --- გამოიყენება კონკრეტული ელემენტის მოსაძებნათ (% --- ნებისმიერი რიცხვი ან ასო, _ --- მხოლოდ ერთი ასო)

DROP TABLE users --- შლის ცხრილს
